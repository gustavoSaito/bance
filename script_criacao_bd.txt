
CREATE DATABASE SLIDE_CAD;


CREATE TABLE USUARIO(
ID_USUARIO INT IDENTITY PRIMARY KEY,
NM_USUARIO VARCHAR(300),
GN_EMAIL VARCHAR(300),
GN_SENHA VARCHAR(100),
GN_SEXO VARCHAR(50),
NR_TELEFONE VARCHAR(50),
GN_ESTADO VARCHAR(200),
GN_CIDADE VARCHAR(200),
FL_ATIVO BIT)

CREATE TABLE LOGIN(
ID_LOGIN INT IDENTITY PRIMARY KEY,
GN_EMAIL VARCHAR(300) NOT NULL,
GN_SENHA VARCHAR(100) NOT NULL,
DT_INCLUSAO DATE,
DT_EXCLUSAO DATE,
FL_ATIVO INT,
ID_USUARIO INT,
CONSTRAINT FK_ID_USUARIO_LOGIN FOREIGN KEY (ID_USUARIO) REFERENCES USUARIO (ID_USUARIO));

CREATE TABLE CLIENTE(
ID_CLIENTE INT IDENTITY PRIMARY KEY,
NM_CLIENTE VARCHAR(200),
GN_EMAIL VARCHAR(200),
GN_DOCUMENTO VARCHAR(200),
NR_TELEFONE VARCHAR(50),
GN_ORIGEM VARCHAR(100),
GN_ESTADO VARCHAR(100),
GN_CIDADE VARCHAR(100),
FL_ATIVO BIT,
ID_USUARIO INT,
CONSTRAINT FK_ID_USUARIO_CLIENTE FOREIGN KEY (ID_USUARIO) REFERENCES USUARIO (ID_USUARIO));
